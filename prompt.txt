nombre: AcaBiblioteca
Tablas{
usuarios: 
id_usuario (PK INT 10 UNIC)
nombre(VARCHAR 100)
apellido (VARCHAR 100)
correo (VARCHAR 250)
tipo_usuario (estudiante, administrador)
fecha_registro. (DATE)
--------------------------------------------------------
libros:
id_libro (PK INT 10 UNIC)
titulo (VARCHAR 150)
autor (VARCHAR 150)
editorial (VARCHAR 150)
anio_publicacion (DATE)
categoria_id (FK → Categorías)
cantidad_disponibles (INT 2)
-------------------------------------------------------
Categorias:
id_categoria (PK INT 10 UNIC)
nombre_categoria (VARCHAR 100)
descripcion (VARCHAR 250)
-------------------------------------------------------
Prestamos:
id_prestamo (PK INT 10 UNIC)
id_usuario (FK → Usuarios)
id_libro (FK → Libros)
fecha_prestamo (date)
fecha_devolucion (DATE)
estado (pendiente, devuelto, retrasado)
------------------------------------------------------
RESERVAS
id_reserva (PK INT 10 UNIC)
id_usuario (FK → Usuarios)
id_libro (FK → Libros)
fecha_reserva (DATE)
estado (activa, cancelada, completada)
------------------------------------------------------
REGISTRO:
id_historial (PK INT 10 UNIC)
id_usuario (FK → Usuarios)
id_libro (FK → Libros)
accion (prestamo, devolucion, reserva)
fecha_accion(DATE)
-----------------------------------------------------
}
